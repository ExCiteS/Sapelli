017    package org.apache.commons.fileupload;
018    
019    
020    /**
021     * The {@link ProgressListener} may be used to display a progress bar
022     * or do stuff like that.
023     */
024    public interface ProgressListener {
025        /** Updates the listeners status information.
026         * @param pBytesRead The total number of bytes, which have been read
027         *   so far.
028         * @param pContentLength The total number of bytes, which are being
029         *   read. May be -1, if this number is unknown.
030         * @param pItems The number of the field, which is currently being
031         *   read. (0 = no item so far, 1 = first item is being read, ...)
032         */
033        void update(long pBytesRead, long pContentLength, int pItems);
034    }
